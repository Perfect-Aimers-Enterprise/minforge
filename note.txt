import torch.nn as nn

class IntentClassifier(nn.Module):
  def __init__(self, input_dim=300, hidden=64, num_classes=4):
    super(IntentClassifier, self)
    nn.Sequential(
        nn.Linear(input_dim, 128),
        nn.ReLu(),
        nn.Dropout(0.3),
        nn.Linear(128, hidden),
        nn.ReLu(),
        nn.Dropout(0.3),
        nn.Linear(hidden, 32),
        nn.ReLu(),
        nn.Dropout(0.3),
        nn.Linear(32, num_classes)
    )
